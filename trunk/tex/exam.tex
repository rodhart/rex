\begin{problem}{Arithmetic}{5}
  \label{prob:arithmetic:prime}
  Which of the following is not prime?

    \answer[correct] $2^{-1}$
    \answer[correct] $2^0$
    \answer $2^1$
    \answer $3^1$
    \answer[fixed] none of the above
  \end{answers}
\end{problem}
